# set custom configs server
export NAMADA_NETWORK_CONFIGS_SERVER="https://testnet.DOMAIN/configs"

# join network
namadac utils join-network --chain-id CHAIN_ID --dont-prefetch-wasm

# download wasm
wget -O - https://testnet.luminara.icu/wasm.tar.gz | tar -xz -C $HOME/.local/share/namada/CHAIN_ID/wasm/ --strip-components=1

# add persistent peer
sed -i 's#persistent_peers = ".*"#persistent_peers = PEER#' $HOME/.local/share/namada/CHAIN_ID/config.toml

# ...and you're ready to start your node!
